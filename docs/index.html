A lightweight composable wrapper for styled-components.

const Button = compose(
  setDisplayName("Button"),
  withStyleAttributes({ type: "button" })
)("button")`
  cursor: pointer;
  background: #1f1f1f;
  color: white;
  border: 0;
  padding: 8px 16px;
  border-radius: 4px;
  text-decoration: none;
  margin: 10px auto;

  &:hover {
    color: #ebebeb;
  }

  &:focus {
    outline: 0;
  }
`;
